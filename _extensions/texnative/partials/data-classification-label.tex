% Data Classification Label
% Renders a colored label box on every page when data_classification_label.level is set

$if(dcl_enabled)$
\usepackage{tikz}
\usepackage{eso-pic}

% Define colors for the classification label
\definecolor{dclbgcolor}{HTML}{$dcl_bgcolor$}
\definecolor{dcltxtcolor}{HTML}{$dcl_txtcolor$}

% Command to draw the classification label
\newcommand{\dataclassificationlabel}{%
	\begin{tikzpicture}[remember picture, overlay]
		\node[
			fill=dclbgcolor,
			text=dcltxtcolor,
			rounded corners=$dcl_border_radius$,
			minimum width=$dcl_width$,
			minimum height=$dcl_height$,
			anchor=north west,
			font=\$dcl_txtsize$
		] at ([xshift=$dcl_left$, yshift=-$dcl_top$]current page.north west) {$dcl_txt$};
	\end{tikzpicture}%
}

% Add the label to every page using eso-pic
\AddToShipoutPictureBG{\dataclassificationlabel}
$endif$
